package com.sd.mobile.data.remote

    val date: String,
    val steps: Int,
    val stress_avg: Double,
    val recommendation: String,
    val reason: String
)

    val items: List<WeeklyItem>
)

data class AckRequest(val date: String)
data class AckResponse(val ok: Boolean, val receivedDate: String?)
